exports.DeleteOneSubmission = async (req, res) => {
    let delid = req.params.id;

    try {
        await SubmissionType.findByIdAndDelete(delid);
        res.send("deleted")
    } catch (err) {
        res.status(500).send({ status: "error in deleting data", error: err.message });
    }
}
